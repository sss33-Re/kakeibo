const SHEET_NAME = '明細'; const FIXED_SHEET_NAME = '固定費'; function doPost(e) { return handle(e); } function doGet(e) { return handle(e); } function handle(e) { try { let data; if (e.postData && e.postData.contents) { data = JSON.parse(e.postData.contents); } else { data = e.parameter; } const ss = SpreadsheetApp.getActiveSpreadsheet(); if (data.action === 'save') { let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) sheet = ss.insertSheet(SHEET_NAME); sheet.clearContents(); sheet.appendRow(['id','date','name','amount','type','cat','payMethod','memo','tags']); const allData = typeof data.allData === 'string' ? JSON.parse(data.allData) : data.allData; allData.forEach(r => { // 日付を必ず文字列として保存（'をつけることで文字列扱いに） sheet.appendRow([ r.id, "'" + String(r.date), // ← 文字列として強制保存 r.name, r.amount, r.type, r.cat, r.payMethod||'カード', r.memo||'', (r.tags||[]).join(',') ]); }); let fsheet = ss.getSheetByName(FIXED_SHEET_NAME); if (!fsheet) fsheet = ss.insertSheet(FIXED_SHEET_NAME); fsheet.clearContents(); fsheet.appendRow(['id','name','amount','type','payMethod']); const fixedList = typeof data.fixedList === 'string' ? JSON.parse(data.fixedList) : (data.fixedList||[]); fixedList.forEach(f => { fsheet.appendRow([f.id, f.name, f.amount, f.type, f.payMethod||'カード']); }); return output({ok: true, count: allData.length}); } if (data.action === 'load') { let sheet = ss.getSheetByName(SHEET_NAME); let allData = []; if (sheet && sheet.getLastRow() > 1) { const rows = sheet.getDataRange().getValues(); allData = rows.slice(1).map(r => { // どんな形式でも正しくYYYY/MM/DDに変換 let dateStr = ''; if (r[1] instanceof Date) { const d = r[1]; dateStr = `${d.getFullYear()}/${String(d.getMonth()+1).padStart(2,'0')}/${String(d.getDate()).padStart(2,'0')}`; } else { // 文字列の場合、先頭の'を除去してハイフンをスラッシュに統一 dateStr = String(r[1]).replace(/^'+/, '').replace(/-/g, '/').trim(); } return { id: String(r[0]), date: dateStr, name: String(r[2]), amount: Number(r[3]), type: String(r[4]), cat: String(r[5]), payMethod: String(r[6])||'カード', memo: String(r[7]), tags: r[8] ? String(r[8]).split(',').filter(Boolean) : [] }; }); } let fsheet = ss.getSheetByName(FIXED_SHEET_NAME); let fixedList = []; if (fsheet && fsheet.getLastRow() > 1) { const frows = fsheet.getDataRange().getValues(); fixedList = frows.slice(1).map(r => ({ id: String(r[0]), name: String(r[1]), amount: Number(r[2]), type: String(r[3]), payMethod: String(r[4])||'カード' })); } return output({ok: true, allData, fixedList}); } return output({ok: false, error: 'unknown action'}); } catch(err) { return output({ok: false, error: err.message}); } } function output(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
